\subsection{Different use cases}

\begin{frame}{Consumer Client}
    \framesubtitle{Different use cases}
    \begin{itemize}
        \item<1-> Visual
        \begin{itemize}
            \item<2-> Display of public transport
            \item<2-> Tracking and information of trucks
        \end{itemize}

        \item<1-> Informative
        \begin{itemize}
            \item<3-> Pizza delivery \& Providing taxi
            \begin{itemize}
                \item Customer notifications of status and time estimates
                \item Helps locating the nearest taxis \& position of driver
            \end{itemize}
        \end{itemize}

        \item<1-> Data Warehouse
        \begin{itemize}
            \item<4-> Business intelligence
            \item<4-> Data analysis
        \end{itemize}
    \end{itemize}

    \note<1-1>[item]{Visual}
    \note<1-1>[item]{Informative}
    \note<1-1>[item]{Data Warehouse}

    \note<2-2>[item]{Public transport: Old, stations transmit location, allows for real--time tracking.}
    \note<2-2>[item]{Trucks: Like our solution, see route, delays, foresee road block}

    \note<3-3>[item]{Pizza delivery:}
    \note<3-3>[item]{Allows for pre--planning of routes}
    \note<3-3>[item]{Give customers notifications about status, estimates and potential delays}
    \note<3-3>[item]{Allows for owner to see status of delivery guy}
    \note<3-3>[item]{}
    \note<3-3>[item]{Providing taxi:}
    \note<3-3>[item]{Helps the company locate the nearest free taxi}
    \note<3-3>[item]{By planning routes from assigned taxi to customer a more precise TTA can be given}
    \note<3-3>[item]{Easier for the company to see when a taxi would be available again}

    \note<4-4>[item]{Data warehouse -> analyze}
    \note<4-4>[item]{emplyee qulity assessment}
    \note<4-4>[item]{business intelligence}
